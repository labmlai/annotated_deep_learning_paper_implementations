53class CIFAR10VGGModel(Module):
# 
58    def conv_block(self, in_channels, out_channels) -> nn.Module:
59        return nn.Sequential(
60            nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1),
61            nn.ReLU(inplace=True),
62        )
# 
64    def __init__(self, blocks: List[List[int]]):
65        super().__init__()
#

5 $2 \times 2$ pooling layers will produce a output of size $1 \ times 1$. CIFAR 10 image size is $32 \times 32$

69        assert len(blocks) == 5
70        layers = []
#

RGB channels

72        in_channels = 3
#

Number of channels in each layer in each block

74        for block in blocks:
#

Convolution, Normalization and Activation layers

76            for channels in block:
77                layers += self.conv_block(in_channels, channels)
78                in_channels = channels
#

Max pooling at end of each block

80            layers += [nn.MaxPool2d(kernel_size=2, stride=2)]
#

Create a sequential model with the layers

83        self.layers = nn.Sequential(*layers)
#

Final logits layer

85        self.fc = nn.Linear(in_channels, 10)
# 
87    def __call__(self, x):
#

The VGG layers

89        x = self.layers(x)
#

Reshape for classification layer

91        x = x.view(x.shape[0], -1)
#

Final linear layer

93        return self.fc(x)
